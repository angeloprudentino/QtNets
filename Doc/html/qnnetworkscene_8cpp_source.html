738 void QNNetworkScene::manageMoveInside(){
739  //do it for every selected item
740  foreach (QGraphicsItemPtr item, this->selectedItems()) {
741  int t = item->type();
742  if(t == QNNodeItem::Type || t == QNModelItem::Type || t == QNTextBoxItem::Type){
743  QNModelItemPtr oldParent = qgraphicsitem_cast<QNModelItemPtr>(item->parentItem());
744  QNModelItemPtr newParent = this->determineParent(item->scenePos(), item);
745  if(newParent != item && newParent != oldParent && (newParent == nullptr || newParent->isShowingContent())){
746  QNModelPtr newModel = (newParent) ? newParent->getModel() : this->network;
747  QNModelPtr oldModel = (oldParent) ? oldParent->getModel() : this->network;
748 
749  //get new position
750  QPointF point;
751  if(newParent != nullptr)
752  point = item->scenePos() - newParent->scenePos();
753  else
754  point = item->scenePos();
755 
756  switch (t) {
757  case QNNodeItem::Type:{
758  QNNodeItemPtr node = qgraphicsitem_cast<QNNodeItemPtr>(item);
759 
760  //move node to new model
761  if(oldModel->moveIcon(node->getNode(), newModel)){
762  //move node item
763  this->removeItem(node);
764  node->setParentItem(newParent);
765  if(newParent == nullptr){
766  this->addItem(node);
767  node->setZValue(0.2);
768  }
769  else
770  node->setZValue(newParent->zValue() + 0.2);
771 
772  QNEdgeItemListPtr edges = node->getEdges();
773  if(edges != nullptr){
774  int size = edges->size();
775  for(int i=0; i<size; i++){

784  e->setParentItem(newParent);
785  }
786  else if(edgeParent != nullptr && edgeNewParent == nullptr){
787  //return to main scene
788  this->removeItem(e);
789  e->setParentItem(newParent);
790  this->addItem(e);
791  }
792  else{
793  //from a model to another
794  e->setParentItem(newParent);
795  }
796 
797  e->setZValue(0.1);
798  e->updateEdgePos();
799  }
800  }
801  }
802 
803  node->setPos(point);
804  }
805 
806  break;
807  }
808  case QNModelItem::Type:{
809  QNModelItemPtr model = qgraphicsitem_cast<QNModelItemPtr>(item);
810 
811  //move model to new model
812  if(oldModel->moveIcon(model->getModel(), newModel)){
813  //move model item

815  model->setParentItem(newParent);
816  if(newParent == nullptr)
817  this->addItem(model);
818 
819  model->setZValue(0.0);
820  model->setPos(point);
821  }
822 
823  break;
824  }
