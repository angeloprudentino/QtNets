67  QPointF dstBR(dstTL.x()+dstW, dstTL.y()+dstH);

77  QPointF srcC(srcTL.x()+srcW/2, srcTL.y()+srcH/2);
78 
79  //center to center line
80  QLineF cToC(dstC, srcC);
81 
82  //determine dst intersection
83  QList<QLineF> dstEdges;
84  dstEdges.push_back(QLineF(dstTL, dstTR)); //top
85  dstEdges.push_back(QLineF(dstBL, dstBR)); //bottom
86  dstEdges.push_back(QLineF(dstTR, dstBR)); //right
87  dstEdges.push_back(QLineF(dstTL, dstBL)); //left
88  QLineF::IntersectType intersectType;

91  intersectType = cToC.intersect(edge, &dstPoint);
92  if (intersectType == QLineF::BoundedIntersection){
93  found = true;

95  }
96  }
97  if(!found)
98  dstPoint = dstBR;

100

102  QList<QLineF> srcEdges;

104  srcEdges.push_back(QLineF(srcBL, srcBR)); //bottom
105  srcEdges.push_back(QLineF(srcTR, srcBR)); //right
106  srcEdges.push_back(QLineF(srcTL, srcBL)); //left
107  found = false;

112  break;
113  }
114  }
