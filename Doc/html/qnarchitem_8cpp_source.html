42 QLineF QNArchItem::determineLine(){
43  //self-loop
44  if(this->srcNode == this->dstNode){
45  return QLineF(this->dstNode->scenePos(), this->srcNode->scenePos());
46  }
47  else{
48  //about dst
49  int dstW = this->dstNode->getWidth();
50  int dstH = this->dstNode->getHeight();
51  QPointF dstTL = this->dstNode->scenePos();
52  QPointF dstBL(dstTL.x(), dstTL.y()+dstH);
53  QPoint dstTR(dstTL.x()+dstW, dstTL.y());
54  QPointF dstBR(dstTL.x()+dstW, dstTL.y()+dstH);
55  QPointF dstC(dstTL.x()+dstW/2, dstTL.y()+dstH/2);
56 
57  //about src
58  int srcW = this->srcNode->getWidth();
59  int srcH = this->srcNode->getHeight();
60  QPointF srcTL = this->srcNode->scenePos();
61  QPointF srcBL(srcTL.x(), srcTL.y()+srcH);
62  QPointF srcTR(srcTL.x()+srcW, srcTL.y());
63  QPointF srcBR(srcTL.x()+srcW, srcTL.y()+srcH);
64  QPointF srcC(srcTL.x()+srcW/2, srcTL.y()+srcH/2);
65 
66  //center to center line
67  QLineF cToC(dstC, srcC);
68 
69  QPointF p1; //dst point
70  QPointF p2; //src point
71 
72  //determine dst intersection
73  QList<QLineF> dstEdges;
74  dstEdges.push_back(QLineF(dstTL, dstTR)); //top
75  dstEdges.push_back(QLineF(dstBL, dstBR)); //bottom
76  dstEdges.push_back(QLineF(dstTR, dstBR)); //right
77  dstEdges.push_back(QLineF(dstTL, dstBL)); //left
78  QLineF::IntersectType intersectType;
79  bool found = false;
80  for(QLineF edge: dstEdges){
81  intersectType = cToC.intersect(edge, &p1);
82  if (intersectType == QLineF::BoundedIntersection){
83  found = true;
84  break;
85  }
86  }
87  if(!found)
88  p1 = dstBR;
89 
90 
91  //determine src intersection
92  QList<QLineF> srcEdges;
93  srcEdges.push_back(QLineF(srcTL, srcTR)); //top
94  srcEdges.push_back(QLineF(srcBL, srcBR)); //bottom
95  srcEdges.push_back(QLineF(srcTR, srcBR)); //right
96  srcEdges.push_back(QLineF(srcTL, srcBL)); //left
97  found = false;
98  for(QLineF edge: srcEdges){
99  intersectType = cToC.intersect(edge, &p2);
100  if (intersectType == QLineF::BoundedIntersection){
101  found = true;
102  break;
103  }
104  }
105  if(!found)
106  p2 = srcTL;
107 
108  return QLineF(p1, p2);
109  }
