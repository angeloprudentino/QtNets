708  QXmlStreamReader xml(&file);
709 
710  //start parsing
711  xml.readNext();
712  while(!xml.atEnd() && !xml.isStartElement()){
713  this->log.d("parse", QString("%1").arg(xml.tokenString()));
714  xml.readNext();
715  }
716

718  this->log.d("parse", QString("%1 tag start").arg(xml.name().toString()));
719  if(xml.name() == STYLE){

721  QString id = xml.attributes().value(STYLE_ID).toString();
722  if(id.isEmpty()){
723  xml.raiseError("style's id must be not null");
724  }
725  else{
726  this->log.d("parse", "style id: " + id);
727 
728  //style version
729  QString version = xml.attributes().value(STYLE_VERSION).toString();
730  if(version.isEmpty()){
731  xml.raiseError("style's version must be not null");
732  }
733  else{
734  this->log.d("parse", "style version: " + version);
735 
736  aStyle.reset(new QNStyle(id, version));
737  if (!xml.hasError()){
738  //parse style's children
739  xml.readNext();
740  this->parseStyleClasses(xml, aStyle.data());
741  }
742  }
743  }
744  }
745  else{
746  xml.raiseError(QString("Root element must be %1").arg(STYLE));
747  }
748  }
749  file.close();
750 
751  if (xml.hasError()){
752  QString err = QString("Failed to parse file %1: %2").arg(file.fileName()).arg(xml.errorString());
753  this->log.e("parse", err);
754  aStyle.reset();
755  return QNError(STYLE_READ_ERROR, err);
756  }
757  else if (file.error() != QFile::NoError){
758  QString err = QString("Cannot read file %1: %2").arg(file.fileName()).arg(file.errorString());
759  this->log.e("parse", err);
760  return QNError(STYLE_READ_ERROR, err);
761  }
762 
763  return QNError();
764 }
